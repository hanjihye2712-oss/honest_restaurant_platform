```python
# honest_restaurant/models.py

from django.db import models


class PublicRestaurantData(models.Model):
    """서울시 공공데이터 - 일반음식점 인허가 정보"""

    # ① 가게 식별 / 기본 정보
    management_no = models.CharField(max_length=50, unique=True, verbose_name="관리번호")
    name          = models.CharField(max_length=200, verbose_name="업소명")
    address_road  = models.CharField(max_length=300, blank=True, verbose_name="도로명주소")
    address_jibun = models.CharField(max_length=300, blank=True, verbose_name="지번주소")
    district      = models.CharField(max_length=20, blank=True, db_index=True, verbose_name="자치구")
    phone         = models.CharField(max_length=30, blank=True, verbose_name="전화번호")
    business_type = models.CharField(max_length=100, blank=True, db_index=True, verbose_name="업태구분명")
    category_name = models.CharField(max_length=100, blank=True, verbose_name="업종명")

    # ② 정부 인증 / 위생 정보
    sanitation_business_type = models.CharField(max_length=100, blank=True, verbose_name="위생업종명")
    license_date             = models.DateField(null=True, blank=True, db_index=True, verbose_name="인허가일자")
    license_cancel_date      = models.DateField(null=True, blank=True, verbose_name="인허가취소일자")

    STATUS_OPEN      = "01"
    STATUS_SUSPENDED = "02"
    STATUS_CLOSED    = "03"
    STATUS_CHOICES   = [("01","영업"), ("02","휴업"), ("03","폐업")]

    status_code      = models.CharField(max_length=10, blank=True, db_index=True,
                                        choices=STATUS_CHOICES, verbose_name="영업상태구분코드")
    area             = models.FloatField(null=True, blank=True, verbose_name="소재지면적(㎡)")
    last_modified_at = models.DateTimeField(null=True, blank=True, verbose_name="최종수정시점")

    # ③ 위치 좌표
    latitude  = models.FloatField(null=True, blank=True, verbose_name="위도")
    longitude = models.FloatField(null=True, blank=True, verbose_name="경도")

    # ④ 동기화 메타
    synced_at  = models.DateTimeField(auto_now=True,     verbose_name="동기화 시각")
    created_at = models.DateTimeField(auto_now_add=True, verbose_name="최초 수집 시각")

    @property
    def is_open(self) -> bool:
        return self.status_code == self.STATUS_OPEN

    @property
    def operating_years(self):
        if not self.license_date:
            return None
        from django.utils import timezone
        return round((timezone.now().date() - self.license_date).days / 365.25, 1)

    @property
    def is_veteran_store(self) -> bool:
        years = self.operating_years
        return years is not None and years >= 10

    def __str__(self):
        return f"[{self.district}] {self.name}"

    class Meta:
        db_table = "public_restaurant_data"
        ordering = ["-synced_at"]
        indexes  = [
            models.Index(fields=["district", "status_code"], name="idx_district_status"),
            models.Index(fields=["latitude", "longitude"],   name="idx_lat_lng"),
        ]


class RestaurantMedia(models.Model):
    """식당 이미지·동영상 파일"""
    TYPE_IMAGE   = "image"
    TYPE_VIDEO   = "video"
    TYPE_CHOICES = [(TYPE_IMAGE, "이미지"), (TYPE_VIDEO, "동영상")]

    restaurant  = models.ForeignKey(PublicRestaurantData, on_delete=models.CASCADE, related_name="media")
    uploaded_by = models.ForeignKey("auth.User", on_delete=models.CASCADE)
    file        = models.FileField(upload_to="restaurant_media/")
    media_type  = models.CharField(max_length=10, choices=TYPE_CHOICES)
    uploaded_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        ordering = ["uploaded_at"]


class ReceiptVerification(models.Model):
    """영수증 인증 신청 — 관리자가 승인/거부"""
    STATUS_PENDING  = "pending"
    STATUS_APPROVED = "approved"
    STATUS_REJECTED = "rejected"
    STATUS_CHOICES  = [("pending","검토 중"), ("approved","인증 완료"), ("rejected","인증 거부")]

    restaurant    = models.ForeignKey(PublicRestaurantData, on_delete=models.CASCADE, related_name="verifications")
    user          = models.ForeignKey("auth.User", on_delete=models.CASCADE)
    status        = models.CharField(max_length=10, choices=STATUS_CHOICES, default="pending")
    receipt_image = models.ImageField(upload_to="receipts/", blank=True, null=True)
    comment       = models.TextField(blank=True)
    submitted_at  = models.DateTimeField(auto_now_add=True)

    class Meta:
        unique_together = ["restaurant", "user"]   # 1인 1건 제한


class Review(models.Model):
    """영수증 인증 완료 사용자만 작성 가능한 리뷰"""
    RATING_CHOICES = [(i, f"{i}점") for i in range(1, 6)]

    restaurant = models.ForeignKey(PublicRestaurantData, on_delete=models.CASCADE, related_name="reviews")
    author     = models.ForeignKey("auth.User", on_delete=models.CASCADE)
    rating     = models.PositiveSmallIntegerField(choices=RATING_CHOICES)
    content    = models.TextField()
    created_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        ordering = ["-created_at"]
```

### 🔑 핵심 포인트

| 코드 | 이유 |
|------|------|
| `unique=True` (management_no) | 같은 가게 중복 저장 방지 |
| `db_index=True` | 필터·검색 성능 향상 |
| `STATUS_OPEN = "01"` 상수 | 하드코딩 방지 → 상수로 관리 |
| `auto_now=True` | 저장할 때마다 현재 시간 자동 기록 |
| `auto_now_add=True` | 최초 생성 시간만 기록, 이후 변경 안 됨 |
| `@property` 3개 | DB 컬럼 없이 계산값 제공 |
| `unique_together` | 한 사용자가 같은 식당에 중복 인증 신청 방지 |
| `related_name` | `restaurant.reviews.all()` 처럼 역참조 가능 |

```python
# honest_restaurant/views.py

from django.shortcuts import render, get_object_or_404
from django.core.paginator import Paginator
from rest_framework import viewsets, filters
from django_filters.rest_framework import DjangoFilterBackend
from .models import PublicRestaurantData
from .serializers import PublicRestaurantDataSerializer


# ── JSON API 뷰셋 ─────────────────────────────────────────
class PublicRestaurantDataViewSet(viewsets.ReadOnlyModelViewSet):
    queryset         = PublicRestaurantData.objects.filter(status_code=PublicRestaurantData.STATUS_OPEN)
    serializer_class = PublicRestaurantDataSerializer
    filter_backends  = [DjangoFilterBackend, filters.SearchFilter, filters.OrderingFilter]
    filterset_fields = ["district", "business_type", "sanitation_business_type"]
    search_fields    = ["name", "address_road", "category_name"]
    ordering_fields  = ["license_date", "name", "synced_at"]
    ordering         = ["-synced_at"]


# ── 메인 인덱스 ───────────────────────────────────────────
def index_page(request):
    return render(request, 'restaurants/index.html')


# ── 식당 목록 (검색·필터·페이지네이션) ────────────────────
def restaurant_list_page(request):
    qs = PublicRestaurantData.objects.filter(status_code=PublicRestaurantData.STATUS_OPEN)

    search        = request.GET.get("search", "").strip()
    district      = request.GET.get("district", "").strip()
    business_type = request.GET.get("business_type", "").strip()

    if search:
        from django.db.models import Q
        qs = qs.filter(
            Q(name__icontains=search) | Q(address_road__icontains=search) |
            Q(address_jibun__icontains=search) | Q(category_name__icontains=search) |
            Q(district__icontains=search) | Q(business_type__icontains=search)
        )
    if district:      qs = qs.filter(district=district)
    if business_type: qs = qs.filter(business_type=business_type)

    paginator   = Paginator(qs.order_by("-synced_at"), 20)
    restaurants = paginator.get_page(request.GET.get("page", 1))

    district_choices = (PublicRestaurantData.objects
        .filter(status_code=PublicRestaurantData.STATUS_OPEN)
        .exclude(district="").values_list("district", flat=True).distinct().order_by("district"))

    business_type_choices = (PublicRestaurantData.objects
        .filter(status_code=PublicRestaurantData.STATUS_OPEN)
        .exclude(business_type="").values_list("business_type", flat=True).distinct().order_by("business_type"))

    query_params = request.GET.copy()
    query_params.pop("page", None)

    return render(request, "restaurants/restaurant_list.html", {
        "restaurants"          : restaurants,
        "total_count"          : paginator.count,
        "district_choices"     : district_choices,
        "business_type_choices": business_type_choices,
        "query_string"         : query_params.urlencode(),
    })


# ── 식당 상세 + 리뷰 처리 ─────────────────────────────────
def restaurant_detail_page(request, pk):
    from .models import Review, ReceiptVerification
    from .forms import ReviewForm
    from django.shortcuts import redirect

    restaurant   = get_object_or_404(PublicRestaurantData, pk=pk)
    reviews      = restaurant.reviews.select_related("author").all()
    has_verified = (
        request.user.is_authenticated and
        ReceiptVerification.objects.filter(
            restaurant=restaurant, user=request.user,
            status=ReceiptVerification.STATUS_APPROVED
        ).exists()
    )

    if request.method == "POST":
        if not request.user.is_authenticated: return redirect("login")
        if not has_verified:
            form = ReviewForm()
        else:
            form = ReviewForm(request.POST)
            if form.is_valid():
                review = form.save(commit=False)
                review.restaurant = restaurant
                review.author     = request.user
                review.save()
                return redirect("public_restaurants:restaurant-detail-page", pk=pk)
    else:
        form = ReviewForm()

    return render(request, "restaurants/restaurant_detail.html", {
        "restaurant": restaurant, "reviews": reviews,
        "form": form, "has_verified": has_verified,
    })


# ── 영수증 인증 ───────────────────────────────────────────
def receipt_verify_page(request, pk):
    from .models import ReceiptVerification
    from .forms import ReceiptVerificationForm
    from django.shortcuts import redirect

    restaurant = get_object_or_404(PublicRestaurantData, pk=pk)
    if not request.user.is_authenticated: return redirect("login")

    existing = ReceiptVerification.objects.filter(restaurant=restaurant, user=request.user).first()

    if request.method == "POST":
        form = ReceiptVerificationForm(request.POST, request.FILES)
        if form.is_valid():
            v = form.save(commit=False)
            v.restaurant = restaurant
            v.user       = request.user
            v.status     = ReceiptVerification.STATUS_PENDING
            if existing:
                existing.receipt_image = v.receipt_image
                existing.comment       = v.comment
                existing.status        = ReceiptVerification.STATUS_PENDING
                existing.save()
            else:
                v.save()
            return redirect("public_restaurants:restaurant-detail-page", pk=pk)
    else:
        form = ReceiptVerificationForm()

    return render(request, "restaurants/restaurant_verify.html", {
        "restaurant": restaurant, "form": form, "existing": existing,
    })


# ── 미디어 업로드 ──────────────────────────────────────────
def media_upload(request, pk):
    from .models import RestaurantMedia
    from .forms import RestaurantMediaForm
    from django.shortcuts import redirect
    import os

    if not request.user.is_authenticated: return redirect("login")
    restaurant = get_object_or_404(PublicRestaurantData, pk=pk)

    if request.method == "POST":
        form = RestaurantMediaForm(request.POST, request.FILES)
        if form.is_valid():
            f          = form.cleaned_data["file"]
            ext        = os.path.splitext(f.name)[1].lower()
            media_type = (RestaurantMedia.TYPE_VIDEO
                          if ext in [".mp4", ".mov", ".avi", ".webm"]
                          else RestaurantMedia.TYPE_IMAGE)
            RestaurantMedia.objects.create(
                restaurant=restaurant, uploaded_by=request.user,
                file=f, media_type=media_type)
    return redirect("public_restaurants:restaurant-detail-page", pk=pk)


# ── 미디어 삭제 ────────────────────────────────────────────
def media_delete(request, pk, media_pk):
    from .models import RestaurantMedia
    from django.shortcuts import redirect
    import os

    if not request.user.is_authenticated: return redirect("login")
    media = get_object_or_404(RestaurantMedia, pk=media_pk, restaurant__pk=pk)
    if media.uploaded_by == request.user or request.user.is_staff:
        if media.file and os.path.isfile(media.file.path):
            os.remove(media.file.path)
        media.delete()
    return redirect("public_restaurants:restaurant-detail-page", pk=pk)
```

## STEP ⑩ 인덱스 검색 → 식당 목록 페이지 연동

### 💡 왜 이 작업이 필요한가?

`views.py`에는 이미 `search`, `district`, `business_type` 쿼리스트링을 처리하는 로직이 있다.
문제는 `index.html`의 검색 폼이 **어디로 데이터를 보낼지** 연결이 안 되어 있었다는 것이다.

```
index.html 검색창  ─→  (연결 없음)                  ← 변경 전
index.html 검색창  ─→  /restaurants/?search=검색어   ← 변경 후
```

또한 **비로그인 사용자**가 검색하면 로그인 페이지로 이동시키는 UX도 추가한다.

### 🔍 동작 원리

```
[검색 버튼 클릭]
       ↓
JS: 로그인 상태 확인 (data-authenticated 속성)
       ↓
  ┌─ 로그인 O ──→ form.submit() → GET /restaurants/?search=검색어
  └─ 로그인 X ──→ /accounts/login/?next=/restaurants/
                  (로그인 후 자동으로 목록 페이지 이동)
```

### 📄 index.html — 검색 폼 수정 부분

```html
<!-- 검색 폼: action으로 목록 페이지 URL 지정 -->
<form
  method="get"
  action="{% url 'public_restaurants:restaurant-list-page' %}"
  class="ph-search"
  id="searchForm"
  data-login-url="{% url 'login' %}"
  data-next-url="{% url 'public_restaurants:restaurant-list-page' %}"
  data-authenticated="{% if request.user.is_authenticated %}true{% else %}false{% endif %}"
>
  <input type="text" name="search" placeholder="동네, 업태, 가게 이름으로 검색"
         value="{{ request.GET.search }}">
  <button type="submit">검색</button>
</form>
```

```javascript
// index.html 하단 <script> 블록
document.getElementById('searchForm').addEventListener('submit', function(e) {
  if (this.dataset.authenticated === 'false') {
    e.preventDefault();
    location.href = this.dataset.loginUrl + '?next=' + this.dataset.nextUrl;
  }
});
```

### 🔑 핵심 포인트

| 항목 | 내용 |
|------|------|
| `action="{% url '...' %}"` | 폼 데이터를 보낼 URL 지정. 없으면 현재 페이지로 전송 |
| `method="get"` | 검색은 GET 방식 → URL에 쿼리스트링 포함 (`?search=김밥`) |
| `data-*` 속성 | Django 템플릿 태그를 JS 속성으로 분리 |
| `?next=` 파라미터 | Django 로그인 뷰가 인식 → 로그인 완료 후 지정 URL로 자동 이동 |
| views.py 수정 불필요 | URL 연결만 해주면 기존 쿼리스트링 처리 로직이 그대로 동작 |

### ⚠️ onsubmit 안에 Django 템플릿 태그 직접 사용 금지

```html
<!-- ❌ 오류: IDE에서 JS 문법 오류 발생 -->
<form onsubmit="location.href='{% url 'login' %}?next=...'">

<!-- ✅ 해결: data-* 속성으로 URL 분리 후 JS에서 참조 -->
<form id="searchForm"
      data-login-url="{% url 'login' %}"
      data-next-url="{% url 'public_restaurants:restaurant-list-page' %}">
```

```python
# honest_restaurant/admin.py

from django.contrib import admin
from .models import PublicRestaurantData, ReceiptVerification, Review


@admin.register(PublicRestaurantData)
class PublicRestaurantDataAdmin(admin.ModelAdmin):
    list_display    = ["name", "district", "business_type", "status_badge", "license_date", "synced_at"]
    list_filter     = ["status_code", "district", "business_type"]
    search_fields   = ["name", "address_road", "management_no"]
    ordering        = ["-synced_at"]
    list_per_page   = 50
    readonly_fields = ["management_no", "synced_at", "created_at", "operating_years", "is_veteran_store", "is_open"]

    @admin.display(description="영업상태")
    def status_badge(self, obj):
        from django.utils.html import format_html
        color_map = {
            "01": ("#d4edda", "#155724", "영업"),
            "02": ("#fff3cd", "#856404", "휴업"),
            "03": ("#f8d7da", "#721c24", "폐업"),
        }
        bg, color, label = color_map.get(obj.status_code, ("#e9ecef", "#495057", obj.status_code))
        # format_html(): HTML 문자열과 삽입값을 반드시 분리 (XSS 방지)
        return format_html(
            '<span style="background:{};color:{};padding:2px 8px;border-radius:999px;font-size:12px;">{}</span>',
            bg, color, label
        )


@admin.register(ReceiptVerification)
class ReceiptVerificationAdmin(admin.ModelAdmin):
    list_display  = ["restaurant", "user", "status", "submitted_at"]
    list_filter   = ["status"]
    search_fields = ["restaurant__name", "user__username"]
    actions       = ["approve", "reject"]

    @admin.action(description="선택 항목 인증 승인")
    def approve(self, request, queryset):
        queryset.update(status=ReceiptVerification.STATUS_APPROVED)

    @admin.action(description="선택 항목 인증 거부")
    def reject(self, request, queryset):
        queryset.update(status=ReceiptVerification.STATUS_REJECTED)


@admin.register(Review)
class ReviewAdmin(admin.ModelAdmin):
    list_display  = ["restaurant", "author", "rating", "created_at"]
    search_fields = ["restaurant__name", "author__username"]
```

## STEP ⑫ accounts 앱 작성

```python
# accounts/views.py

from django.shortcuts import render
from django.contrib.auth.forms import UserCreationForm
from django.http import HttpResponseRedirect
from django.urls import reverse

def signup(request):
    if request.method == "POST":
        form = UserCreationForm(request.POST)
        if form.is_valid():
            form.save()
            return HttpResponseRedirect(reverse("login"))
    else:
        form = UserCreationForm()
    return render(request, "accounts/signup.html", {"form": form})
```

## 🐛 실제 발생한 오류 4가지

### 오류 1 — NoReverseMatch (네임스페이스 누락)

```
NoReverseMatch: Reverse for 'restaurant-detail-page' not found.
```

**원인**: `app_name = "public_restaurants"` 설정 후 템플릿에서 앱 이름 없이 URL 참조

```html
<!-- ❌ -->  {% url 'restaurant-detail-page' r.pk %}
<!-- ✅ -->  {% url 'public_restaurants:restaurant-detail-page' r.pk %}
```

---

### 오류 2 — Django 5+ 로그아웃 GET 거부

**원인**: Django 5 이상에서 `LogoutView`는 보안상 GET 요청을 거부

```html
<!-- ❌ -->  <a href="/accounts/logout/">로그아웃</a>
<!-- ✅ -->  <form method="post" action="/accounts/logout/" style="display:contents;">
              {% csrf_token %}
              <button type="submit">로그아웃</button>
            </form>
```

---

### 오류 3 — 존재하지 않는 템플릿 상속

**원인**: 없는 앱의 템플릿을 extends로 상속

```html
<!-- ❌ -->  {% extends "polls/base.html" %}
<!-- ✅ -->  {% extends "base.html" %}
```

---

### 오류 4 — onsubmit 안에 Django 템플릿 태그

**원인**: HTML 속성 안의 `{% url %}` 태그를 IDE가 JS 문법으로 파싱해 오류

```html
<!-- ❌ IDE 문법 오류 -->
<form onsubmit="location.href='{% url 'login' %}'">

<!-- ✅ data-* 속성으로 분리 -->
<form data-login-url="{% url 'login' %}" id="myForm">
<script>
  document.getElementById('myForm').addEventListener('submit', function(e) {
    location.href = this.dataset.loginUrl;
  });
</script>
```

## 💬 핵심 교훈 9가지

1. **모델 → 시리얼라이저 → 뷰 → URL** 순서가 Django 개발의 기본 흐름이다
2. **app_name 네임스페이스** 설정 시 템플릿에서 반드시 `앱이름:URL이름` 형식을 써야 한다
3. **Django 5+에서 로그아웃은 POST** — `<a>` 링크(GET)로 처리하면 오류 발생
4. **`{% extends %}`에서 존재하지 않는 템플릿을 상속**하면 TemplateDoesNotExist 오류 발생
5. **CSS 명시도(Specificity)**: `.hdr-nav a`가 `.hdr-login`보다 구체적 → `:not()` 으로 해결
6. **ImageField 사용 시 Pillow 설치 필수** — 없으면 마이그레이션 단계에서 오류
7. **시리얼라이저는 JSON API 전용** — 템플릿 뷰에서는 모델 객체를 직접 사용하므로 불필요
8. **CSS는 HTML에서 분리** — `<style>` 태그 대신 별도 `.css` 파일로 관리해야 유지보수가 쉽다
9. **Django 템플릿 태그는 JS 속성 안에 직접 쓰지 말 것** — `data-*` 속성으로 분리 후 JS에서 참조
